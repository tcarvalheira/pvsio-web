Module: Tab

Renders a Tab or Pills container Content in each panel should be inserted in {tab-widget-id}-{page-id}-pane e.g

Author:

Example

Example use of the widget.

 // Example pvsio-web demo that uses Tab
 // The following configuration assumes the pvsio-web demo is stored in a folder within pvsio-web/examples/demo/
 require.config({
     baseUrl: "../../client/app",
     paths: { d3: "../lib/d3", text: "../lib/text" }
 });
 require(["widgets/container/Tab"], function (Tab) {
      "use strict";
      var disp = new Tab('nav_pill',
                {top: 500,left: 50,width: 800,height: 50},
                {
                    parent: 'content',
                    pages: [{ id: 'battery', title: 'Battery level', state: 'BATTERY' }, 
                            { id: 'date', title: 'Date', state: 'DATE' }, 
                            { id: 'ok', title: 'Ok, go on!', state: 'OK' }, 
                            { id: 'cancel', title: 'Cancel that!!', state: 'CANCEL' }],
                    type: 'pill',
                    opacity: '0',
                    callback: onMessageReceived,
                    displayKey: 'mode'
                })
      disp.render();
 });

Properties
Name Type Attributes Default Description
type 'tab' | 'pills' <optional>
 'tab' whether if container should be a tab ou a pills navigator
pages array.<Objects> <optional>
 [] these objects should have the following schema {id: ... , title: ..., state: ...} state should match with PVS states
callback function <optional>
 id callback function to pass into widgets.
displayKey String <optional>
 'mode' Key that holds active carousel page. PVS state must have this name.
visibleWhen String <optional>
 true Set when widget should or should not be displayed. This can be an expression to match with PVS state.
